public int lengthofLongestSubstring(String s){
    Set<Character> set = new HashSet<>();
    int n = s.length();
    int ans = 0, rk = -1;
    for(int i=0; i<n; i++){
        if(i>0){
            set.remove(s.charAt(i-1));
        }
        while(rk+1<n && set.contains(s.charAt(rk+1))){
            set.add(++rk);
        }
        ans = Math.max(ans, rk-i+1);
    }
    return ans;
}
